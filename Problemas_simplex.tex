\section*{Problema 1: Tipo Unica}
Maximizar: $Z = 7x_1 + 1x_2$

Sujeto a:
\begin{align*}
9x_1 + 7x_2 &\leq 26 \\
6x_1 + 7x_2 &\leq 22 \\
5x_1 + 3x_2 &\leq 34 \\
x_i &\geq 0
\end{align*}
\vspace{1cm}
\section*{Problema 2: Tipo Sin_solucion}
Maximizar: $Z = 3x_1 + 2x_2$

Sujeto a:
\begin{align*}
1.0x_1 + 0.0x_2 &\leq 5.0 \\
-1.0x_1 + 0.0x_2 &\leq -10.0 \\
4.0x_1 + 6.0x_2 &\leq 15.0 \\
x_i &\geq 0
\end{align*}
\vspace{1cm}
\section*{Problema 3: Tipo Acotada}
Maximizar: $Z = 3x_1 + 1x_2$

Sujeto a:
\begin{align*}
3x_1 + 9x_2 &\leq 47 \\
9x_1 + 13x_2 &\leq 64 \\
8x_1 + 13x_2 &\leq 74 \\
x_i &\geq 0
\end{align*}
\vspace{1cm}
\section*{Problema 4: Tipo Unica}
Maximizar: $Z = 4x_1 + 9x_2$

Sujeto a:
\begin{align*}
7x_1 + 8x_2 &\leq 30 \\
6x_1 + 3x_2 &\leq 18 \\
5x_1 + 3x_2 &\leq 16 \\
x_i &\geq 0
\end{align*}
