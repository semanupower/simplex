\section*{Problema 1: Tipo: unica}
Maximizar: $Z = 7x_1 + 8x_2$

Sujeto a:
\begin{align*}
8x_1 + 3x_2 &\leq 36 \\
5x_1 + 1x_2 &\leq 15 \\
4x_1 + 7x_2 &\leq 13 \\
x_1, x_2 &\geq 0
\end{align*}
\vspace{1cm}
\section*{Problema 2: Tipo: sin solucion}
Maximizar: $Z = 2x_1 + 8x_2$

Sujeto a:
\begin{align*}
1.0x_1 + 0.0x_2 &\leq 5.0 \\
-1.0x_1 + 0.0x_2 &\leq -10.0 \\
3.0x_1 + 3.0x_2 &\leq 12.0 \\
x_1, x_2 &\geq 0
\end{align*}
\vspace{1cm}
\section*{Problema 3: Tipo: acotada}
Maximizar: $Z = 2x_1 + 6x_2$

Sujeto a:
\begin{align*}
12x_1 + 8x_2 &\leq 43 \\
3x_1 + 3x_2 &\leq 67 \\
9x_1 + 10x_2 &\leq 33 \\
x_1, x_2 &\geq 0
\end{align*}
